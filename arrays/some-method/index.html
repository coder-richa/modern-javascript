        const createUsername = (user) => {
        let words = user.name.toLowerCase().split(" ");
        // Map creates a new array with the initials which can be used further to create username
        let initials = words
            .map(function (word) {
            return word[0];
            })
            .join("");
        user.username = initials;
        };
